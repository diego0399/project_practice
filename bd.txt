CREATE TABLE ESTUDIANTES(
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre VARCHAR2(200),
    edad INTEGER,
    ciudad NUMBER,
    
    CONSTRAINT FK_ESTUDIANTES_CIUDAD 
        FOREIGN KEY (ciudad) 
        REFERENCES CIUDAD(id)
